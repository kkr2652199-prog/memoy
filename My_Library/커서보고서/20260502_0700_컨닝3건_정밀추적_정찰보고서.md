### 1-A SQL — 정렬 일치 · 미래 `draw_no` ≤ 1221

```sql
SELECT p.brain_tag, p.target_draw_no, p.num1, p.num2, p.num3, p.num4, p.num5, p.num6,
       p.matched_count, p.created_at,
       d.draw_no AS future_draw, d.num1 AS f1, d.num2 AS f2, d.num3 AS f3,
       d.num4 AS f4, d.num5 AS f5, d.num6 AS f6, d.bonus AS fbonus, d.draw_date
FROM lotto_predictions_army3 p
JOIN lotto_draws d
  ON p.num1 = d.num1 AND p.num2 = d.num2 AND p.num3 = d.num3
 AND p.num4 = d.num4 AND p.num5 = d.num5 AND p.num6 = d.num6
WHERE p.target_draw_no BETWEEN 7 AND 1221
  AND d.draw_no > p.target_draw_no
  AND d.draw_no <= 1221
ORDER BY p.target_draw_no, p.brain_tag, d.draw_no;
```

```sql
SELECT p.brain_tag, p.target_draw_no,
       p.num1||','||p.num2||','||p.num3||','||p.num4||','||p.num5||','||p.num6 AS pred_set,
       d.draw_no AS future_draw,

  AND (p.num1 IN (d.num1,d.num2,d.num3,d.num4,d.num5,d.num6))
  AND (p.num2 IN (d.num1,d.num2,d.num3,d.num4,d.num5,d.num6))
```
