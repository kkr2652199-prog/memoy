```12:35:d:\MONEY lol\My_Library\app\lotto\predict_markov.py
def build_transition_matrix(draws: list[dict], decay: float = 0.02) -> dict:
    """
    연속 회차 간 전이행렬 구축.
    draws는 draw_no 오름차순 정렬된 리스트.
    decay: 최신 회차에 높은 가중치 (지수감쇠).

    전이행렬[a][b] = "a가 나온 회차 다음에 b가 나올 가중 횟수"
    """
    matrix = {}
    for i in range(1, 46):
        matrix[i] = {}
        for j in range(1, 46):
            matrix[i][j] = 0.5  # Laplace smoothing (k=0.5)

    total_draws = len(draws)
    for idx in range(len(draws) - 1):
        current_nums = [draws[idx][f"num{k}"] for k in range(1, 7)]
        next_nums = [draws[idx + 1][f"num{k}"] for k in range(1, 7)]
        weight = exp(-decay * (total_draws - 1 - idx))
        for a in current_nums:
            for b in next_nums:
                matrix[a][b] += weight

    return matrix
```

```97:114:d:\MONEY lol\My_Library\app\lotto\predict_lstm.py
    def _build_xy_tensors(
        draws: list[dict], device: Any
    ) -> tuple[Any, Any, int]:
        """(N, SEQ_LEN, 45), (N, 45) float, 샘플 수 n."""
        n = len(draws) - SEQ_LEN
        if n <= 0:
            raise ValueError("학습쌍 0")
        xs: list[list[list[float]]] = []
        ys: list[list[float]] = []
        for k in range(n):
            block = draws[k : k + SEQ_LEN + 1]
            seq = [_multihot_from_draw(block[j]) for j in range(SEQ_LEN)]
            target = _multihot_from_draw(block[SEQ_LEN])
            xs.append(seq)
            ys.append(target)
        x_t = torch.tensor(xs, dtype=torch.float32, device=device)  # type: ignore[union-attr]
        y_t = torch.tensor(ys, dtype=torch.float32, device=device)  # type: ignore[union-attr]
        return x_t, y_t, n
```

```227:284:d:\MONEY lol\My_Library\app\lotto3\predict_contrarian.py
def army3_contrarian_predict(
    training_draws: list[dict],
    target_draw_no: int,
    n_sets: int = 5,
) -> list[dict]:
    if not training_draws or n_sets <= 0:
        return []
    pair_scores = _calculate_d_pair_scores(training_draws, target_draw_no)
    gap_w = _calculate_gap10_weights(training_draws, target_draw_no)
    pmf = _pmf_from_pairs(pair_scores, gap_w)
    nums = list(range(1, 46))
    weights = [pmf[n] for n in nums]
    army_sets = _load_army_sets_for_target(target_draw_no)

    tail = training_draws[-min(10, len(training_draws)) :]
    key_tuples: list[tuple[int, ...]] = []
    for d in tail:
        dn = int(d.get("draw_no", 0) or 0)
        ns = _draw_nums(d)
        if ns:
            key_tuples.append((dn, *ns))
    rng = random.Random(hash(tuple(key_tuples)) % (2**32))

    out: list[dict] = []
    attempts = 0
    max_attempts = 2500
    used: set[tuple[int, ...]] = set()
    while len(out) < n_sets and attempts < max_attempts:
        attempts += 1
        cand = sorted(rng.choices(nums, weights=weights, k=6))
        if len(set(cand)) != 6:
            continue
        try:
            if not tier1_filter(cand):
                continue
        except (TypeError, ValueError):
            continue
        if not _sum_odd_ok(cand):
            continue
        if not _is_diff_from_armies(cand, army_sets, 0.4):
            continue
        t = tuple(cand)
        if t in used:
            continue
        used.add(t)
        pb = _pair_boost_score(cand, pair_scores)
        out.append(
            {
                "nums": cand,
                "brain_tag": "v12_contrarian",
                "method": "🦅 역발상가",
                "confidence": round(min(0.85, 0.35 + 0.02 * math.log1p(pb)), 2),
                "reasoning": f"D-pair+gap10, pair_boost={pb:.1f}, try={attempts}",
            }
        )
    if len(out) < n_sets:
        logger.warning("contrarian: only %d/%d sets for target=%d", len(out), n_sets, target_draw_no)
    return out
```
