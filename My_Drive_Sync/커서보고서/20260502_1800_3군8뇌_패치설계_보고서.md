```258:274:d:\MONEY lol\My_Library\app\lotto\feedback.py
    """한국 로또 등수 기반 점수 계산.
```

```183:193:d:\MONEY lol\My_Library\app\lotto3\predict_contrarian.py
def _is_diff_from_armies(combo: list[int], army_sets: list[set[int]], threshold: float = 0.4) -> bool:
    if not army_sets:
        return True
    s = set(combo)
    for a in army_sets:
        inter = len(s & a)
        uni = len(s | a)
        jac = inter / uni if uni else 0.0
        if jac >= threshold:
            return False
    return True
```
